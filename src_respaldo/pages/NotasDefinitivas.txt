import { useState, useEffect } from 'react';
import { Lock, CheckCircle, TrendingUp, Calculator, Award } from 'lucide-react';
import { motion } from 'framer-motion';

const API = 'http://192.168.10.3:3000';

const NOMBRES_GRADOS: Record<string, string> = {
  '16': '1er Año', '17': '2do Año', '18': '3er Año', '19': '4to Año', '20': '5to Año',
};

const GRADOS_IDS: Record<string, number> = {
  '1er Año': 16, '2do Año': 17, '3er Año': 18, '4to Año': 19, '5to Año': 20,
};

interface NotaDefinitiva {
  estudiante_id: number;
  estudiante_nombre: string;
  estudiante_apellido: string;
  cedula: string;
  promedio_cuantitativo: number | null;
  promedio_cualitativo: number | null;
  definitiva_momento: number | null;
  aprobado: boolean;
}

export default function NotasDefinitivas({ user, cedula }: { user?: any; cedula?: string }) {
  const getToken = () => localStorage.getItem('siga_token') || '';
  const authHeaders = getToken() ? { 'Authorization': `Bearer ${getToken()}`, 'Content-Type': 'application/json' } : {};

  const [loading, setLoading] = useState(true);
  const [grados, setGrados] = useState<string[]>([]);
  const [grado, setGrado] = useState<string>('5to Año');
  const [seccion, setSeccion] = useState<string>('A');
  const [lapso, setLapso] = useState<number>(1);
  const [notas, setNotas] = useState<NotaDefinitiva[]>([]);
  const [isClosed, setIsClosed] = useState(false);
  const [estadisticas, setEstadisticas] = useState({ aprobados: 0, reprobados: 0, promedioGeneral: 0 });

  // Cargar grados disponibles
  useEffect(() => {
    const fetchGrados = async () => {
      try {
        const res = await fetch(`${API}/api/grados`, { headers: authHeaders });
        const data = await res.json();
        setGrados(data.map((g: any) => NOMBRES_GRADOS[String(g.id)] || g.nombre));
      } catch (err) {
        // Fallback manual
        setGrados(['1er Año', '2do Año', '3er Año', '4to Año', '5to Año']);
      }
    };
    fetchGrados();
  }, []);

  // Cargar notas definitivas
  useEffect(() => {
    if (!grado || !seccion) return;

    const fetchNotas = async () => {
      setLoading(true);
      try {
        const gradoId = GRADOS_IDS[grado];
        
        // Obtener estudiantes
        const estRes = await fetch(`${API}/api/estudiantes/${gradoId}/${seccion}`, { headers: authHeaders });
        const estudiantes = await estRes.json();
        const sortedEst = Array.isArray(estudiantes) ? [...estudiantes].sort((a, b) => {
          const numA = parseInt(String(a.cedula).replace(/\D/g, ''), 10) || 0;
          const numB = parseInt(String(b.cedula).replace(/\D/g, ''), 10) || 0;
          return numA - numB;
        }) : [];

        // Obtener promedios cuantitativos y cualitativos por estudiante
        const notasData = await Promise.all(
          sortedEst.map(async (est: any) => {
            // Promedio cuantitativo
            const cuantRes = await fetch(`${API}/api/definitivas/cuantitativo?estudiante_id=${est.id}&lapso=${lapso}`, { headers: authHeaders });
            const cuantData = await cuantRes.json();
            
            // Promedio cualitativo
            const cualRes = await fetch(`${API}/api/definitivas/cualitativo?estudiante_id=${est.id}&lapso=${lapso}`, { headers: authHeaders });
            const cualData = await cualRes.json();

            const promCuanti = cuantData.promedio || null;
            const promCuali = cualData.promedio || null;
            
            // Fórmula: Definitiva = (Cuantitativo × 0.7) + (Cualitativo × 0.3)
            let definitiva = null;
            if (promCuanti !== null && promCuali !== null) {
              definitiva = (promCuanti * 0.7) + (promCuali * 0.3);
            } else if (promCuanti !== null) {
              definitiva = promCuanti;
            } else if (promCuali !== null) {
              definitiva = promCuali;
            }

            return {
              estudiante_id: est.id,
              estudiante_nombre: est.nombres,
              estudiante_apellido: est.apellidos,
              cedula: est.cedula,
              promedio_cuantitativo: promCuanti,
              promedio_cualitativo: promCuali,
              definitiva_momento: definitiva ? Math.round(definitiva * 100) / 100 : null,
              aprobado: definitiva !== null && definitiva >= 10
            };
          })
        );

        setNotas(notasData);

        // Calcular estadísticas
        const aprobados = notasData.filter(n => n.aprobado).length;
        const reprobados = notasData.filter(n => !n.aprobado && n.definitiva_momento !== null).length;
        const promediosValidos = notasData.filter(n => n.definitiva_momento !== null).map(n => n.definitiva_momento!);
        const promedioGeneral = promediosValidos.length > 0 
          ? promediosValidos.reduce((a, b) => a + b, 0) / promediosValidos.length 
          : 0;

        setEstadisticas({ aprobados, reprobados, promedioGeneral: Math.round(promedioGeneral * 100) / 100 });

        // Verificar si el momento está cerrado
        const cierreRes = await fetch(`${API}/api/momentos/cerrado?lapso=${lapso}`, { headers: authHeaders });
        const { cerrado } = await cierreRes.json();
        setIsClosed(cerrado);

      } catch (err) {
        console.error('Error cargando notas definitivas:', err);
      } finally {
        setLoading(false);
      }
    };

    fetchNotas();
  }, [grado, seccion, lapso]);

  const formatearNota = (nota: number | null) => {
    if (nota === null) return '—';
    // Para mostrar: si es entero, sin decimales; si no, con 2 decimales
    return Number.isInteger(nota) ? nota.toString() : nota.toFixed(2);
  };

  const getColorNota = (nota: number | null) => {
    if (nota === null) return 'text-slate-400';
    if (nota >= 18) return 'text-emerald-600';
    if (nota >= 15) return 'text-emerald-500';
    if (nota >= 10) return 'text-yellow-600';
    return 'text-red-600';
  };

  if (loading) {
    return <div className="p-10 text-slate-400 font-bold text-center animate-pulse">Cargando cuadro de notas definitivas...</div>;
  }

  return (
    <div className="min-h-screen bg-slate-50 p-4 md:p-8 max-w-[1600px] mx-auto pb-24">
      
      {/* HEADER */}
      <div className="bg-white rounded-[2rem] md:rounded-[2.5rem] px-6 md:px-10 py-6 md:py-8 mb-8 shadow-sm border-b-[8px] border-blue-600 flex flex-col lg:flex-row justify-between items-start lg:items-center gap-6">
        <div>
          <h1 className="text-2xl md:text-4xl font-black text-slate-900 uppercase tracking-tighter italic leading-none">Notas Definitivas por Momento</h1>
          <p className="text-[10px] font-bold text-blue-700 uppercase tracking-[0.2em] mt-3">
            Fórmula: (Cuantitativo × 70%) + (Cualitativo × 30%)
          </p>
        </div>
        {isClosed && (
          <div className="flex items-center gap-2 px-4 py-2 rounded-2xl bg-red-50 border border-red-200">
            <Lock className="w-4 h-4 text-red-600" />
            <span className="text-[9px] font-black text-red-600 uppercase">Momento Cerrado</span>
          </div>
        )}
      </div>

      {/* TARJETAS DE ESTADÍSTICAS */}
      <div className="grid grid-cols-1 md:grid-cols-4 gap-4 md:gap-6 mb-8">
        <div className="bg-gradient-to-r from-emerald-500 to-emerald-600 rounded-[1.5rem] p-5 text-white shadow-lg">
          <div className="flex items-center justify-between">
            <div>
              <p className="text-[9px] font-black uppercase opacity-80">Aprobados</p>
              <p className="text-3xl font-black">{estadisticas.aprobados}</p>
            </div>
            <Award className="w-8 h-8 opacity-80" />
          </div>
        </div>
        <div className="bg-gradient-to-r from-red-500 to-red-600 rounded-[1.5rem] p-5 text-white shadow-lg">
          <div className="flex items-center justify-between">
            <div>
              <p className="text-[9px] font-black uppercase opacity-80">Reprobados</p>
              <p className="text-3xl font-black">{estadisticas.reprobados}</p>
            </div>
            <TrendingUp className="w-8 h-8 opacity-80" />
          </div>
        </div>
        <div className="bg-gradient-to-r from-blue-500 to-blue-600 rounded-[1.5rem] p-5 text-white shadow-lg col-span-2">
          <div className="flex items-center justify-between">
            <div>
              <p className="text-[9px] font-black uppercase opacity-80">Promedio General del Grupo</p>
              <p className="text-3xl font-black">{formatearNota(estadisticas.promedioGeneral)} pts</p>
            </div>
            <Calculator className="w-8 h-8 opacity-80" />
          </div>
        </div>
      </div>

      {/* SELECTORES */}
      <div className="bg-white rounded-[2rem] p-6 md:p-8 shadow-sm border border-slate-200 mb-8">
        <div className="grid grid-cols-1 md:grid-cols-4 gap-6">
          {/* Lapso */}
          <div>
            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.15em] mb-3">Momento</p>
            <div className="flex gap-2">
              {[1, 2, 3].map(m => (
                <button key={m} onClick={() => setLapso(m)}
                  className={`flex-1 py-3 rounded-2xl font-black text-[11px] uppercase tracking-widest transition-all border-2
                    ${lapso === m ? 'bg-slate-900 text-white border-slate-900' : 'bg-slate-50 text-slate-400 border-slate-100'}`}>
                  {m}°
                </button>
              ))}
            </div>
          </div>

          {/* Grado */}
          <div>
            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.15em] mb-3">Grado / Año</p>
            <select 
              value={grado} 
              onChange={(e) => setGrado(e.target.value)}
              className="w-full p-3 rounded-xl border-2 border-slate-200 font-black text-sm bg-white"
            >
              {grados.map(g => <option key={g} value={g}>{g}</option>)}
            </select>
          </div>

          {/* Sección */}
          <div>
            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.15em] mb-3">Sección</p>
            <div className="flex gap-2">
              {['A', 'B', 'C', 'D', 'E'].map(s => (
                <button key={s} onClick={() => setSeccion(s)}
                  className={`px-4 py-2 rounded-xl font-black text-sm ${seccion === s ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-400'}`}>
                  {s}
                </button>
              ))}
            </div>
          </div>

          {/* Info */}
          <div className="flex items-end">
            <div className="bg-slate-100 rounded-xl p-3 w-full">
              <p className="text-[8px] font-black text-slate-400 uppercase">Total Estudiantes</p>
              <p className="text-xl font-black text-slate-800">{notas.length}</p>
            </div>
          </div>
        </div>
      </div>

      {/* TABLA DE NOTAS DEFINITIVAS */}
      <div className="bg-white rounded-[2.5rem] shadow-sm border border-slate-200 overflow-hidden">
        <div className="px-6 py-5 bg-slate-900 flex justify-between items-center">
          <h3 className="font-black text-white uppercase text-sm">
            {grado} - Sección {seccion} — {lapso}° Momento (Notas Definitivas)
          </h3>
          <div className="flex gap-4 text-[9px]">
            <span className="text-emerald-400">● Aprobado (≥10)</span>
            <span className="text-red-400">● Reprobado (&lt;10)</span>
          </div>
        </div>

        <div className="overflow-x-auto">
          <table className="w-full min-w-[800px]">
            <thead className="bg-slate-50 border-b-2 border-slate-100">
              <tr>
                <th className="px-6 py-4 text-left text-[10px] font-black text-slate-400 w-12">#</th>
                <th className="px-6 py-4 text-left text-[10px] font-black text-slate-400">Estudiante</th>
                <th className="px-6 py-4 text-center text-[10px] font-black text-slate-400 bg-emerald-50">
                  Cuantitativo<br/><span className="text-[8px]">(70%)</span>
                </th>
                <th className="px-6 py-4 text-center text-[10px] font-black text-slate-400 bg-purple-50">
                  Cualitativo<br/><span className="text-[8px]">(30%)</span>
                </th>
                <th className="px-6 py-4 text-center text-[10px] font-black text-slate-400 bg-blue-100">
                  Definitiva<br/><span className="text-[8px]">(100%)</span>
                </th>
                <th className="px-6 py-4 text-center text-[10px] font-black text-slate-400">Estado</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-slate-100">
              {notas.length === 0 ? (
                <tr>
                  <td colSpan={6} className="px-8 py-20 text-center font-black text-slate-300 uppercase italic">
                    No hay estudiantes en esta sección
                  </td>
                </tr>
              ) : (
                notas.map((nota, idx) => (
                  <tr key={nota.estudiante_id} className="hover:bg-slate-50 transition-colors">
                    <td className="px-6 py-4 font-mono font-black text-slate-300 text-xs">{idx + 1}</td>
                    <td className="px-6 py-4">
                      <p className="font-bold text-sm">{nota.estudiante_apellido}, {nota.estudiante_nombre}</p>
                      <p className="text-[9px] text-slate-400">V-{nota.cedula}</p>
                    </td>
                    <td className="px-6 py-4 text-center">
                      <span className={`font-black text-lg ${getColorNota(nota.promedio_cuantitativo)}`}>
                        {formatearNota(nota.promedio_cuantitativo)}
                      </span>
                    </td>
                    <td className="px-6 py-4 text-center">
                      <span className={`font-black text-lg ${getColorNota(nota.promedio_cualitativo)}`}>
                        {formatearNota(nota.promedio_cualitativo)}
                      </span>
                    </td>
                    <td className="px-6 py-4 text-center">
                      <span className={`text-xl font-black ${nota.aprobado ? 'text-emerald-600' : 'text-red-600'}`}>
                        {formatearNota(nota.definitiva_momento)}
                      </span>
                    </td>
                    <td className="px-6 py-4 text-center">
                      {nota.definitiva_momento !== null ? (
                        <span className={`px-3 py-1 rounded-full text-[9px] font-black uppercase ${
                          nota.aprobado ? 'bg-emerald-100 text-emerald-700' : 'bg-red-100 text-red-700'
                        }`}>
                          {nota.aprobado ? 'Aprobado' : 'Reprobado'}
                        </span>
                      ) : (
                        <span className="text-[8px] font-black text-slate-300 uppercase">Sin notas</span>
                      )}
                    </td>
                  </tr>
                ))
              )}
            </tbody>
          </table>
        </div>

        {/* PIE DE TABLA CON FÓRMULA */}
        <div className="px-6 py-4 bg-slate-50 border-t border-slate-100">
          <div className="flex justify-center items-center gap-6 text-[9px] text-slate-500">
            <div className="flex items-center gap-2">
              <div className="w-3 h-3 rounded-full bg-emerald-500"></div>
              <span>Cuantitativas: Evaluaciones planificadas</span>
            </div>
            <div className="flex items-center gap-2">
              <div className="w-3 h-3 rounded-full bg-purple-500"></div>
              <span>Cualitativas: Criterios del docente</span>
            </div>
            <div className="flex items-center gap-2">
              <Calculator className="w-3 h-3" />
              <span>Definitiva = (Cuanti × 0.7) + (Cuali × 0.3)</span>
            </div>
          </div>
        </div>
      </div>

      {/* NOTA ACLARATORIA */}
      {!isClosed && (
        <div className="mt-6 p-4 bg-yellow-50 rounded-2xl border border-yellow-200">
          <p className="text-[9px] text-yellow-700 text-center font-bold uppercase tracking-widest">
            ⚠️ Estas notas son provisionales. El momento debe ser cerrado por Control de Estudios para su publicación oficial.
          </p>
        </div>
      )}
    </div>
  );
}
